int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n,r;
        cin>>n>>r;
        char ch[n][n];
        for(int i=0; i<n; i++)
        {
            for(int j=0; j<n; j++)
            {
                ch[i][j] = '.';
            }
        }
        if(n==r && n!=1 && r!=1)
        {
            cout<<-1<<endl;
        }
        else if(n%r>2)cout<<-1<<endl;
        //else if((n/2)<r)cout<<-1<<endl;
        else
        {
            int i=0,j=0;
            while(r--)
            {
                ch[i][j] = 'R';
                i+=2;
                j+=2;
            }
            for(int i=0; i<n; i++)
            {
                for(int j=0; j<n; j++)
                {
                    cout<<ch[i][j];
                }
                cout<<endl;
            }
        }
    }
}
